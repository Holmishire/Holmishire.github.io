await fetch(rootPath + '/lists/all-cards.json')
					.then(response => response.json())
					.then(json => {
						card_list = json;
				}).catch(error => console.error('Error:', error));

				await fetch(rootPath + '/resources/replacechars.txt')
					.then(response => response.text())
					.then(text => {
						specialchars = text; 
				}).catch(error => console.error('Error:', error));

				card_list_arrayified = card_list.cards;

				let tagCategories = {}; // { category: Set(...) }
				card_list_arrayified.forEach(card => {
					if (card.notes) {
						const tagMatches = card.notes.matchAll(/!tag(?:<(\w+)>)?\s+([^\s\n\b]+)/g);
						
						for (const match of tagMatches) {
							const category = match[1] || "tag"; // default category
							const value = match[2];

							if (!tagCategories[category]) {
								tagCategories[category] = new Set();
							}
							tagCategories[category].add(value);
						}
					}
				});

				const searchInput = document.getElementById("search");
				const tagDropdown = document.getElementById("tag-dropdown");

				// Create a mirror element for position calculation
				const mirror = document.createElement("div");
				Object.assign(mirror.style, {
					position: "absolute",
					visibility: "hidden",
					whiteSpace: "pre",
					font: getComputedStyle(searchInput).font,
					padding: "0", // No padding for pure text measurement
					border: "none",
					boxSizing: "content-box"
				});
				document.body.appendChild(mirror);

				if (searchInput && tagDropdown) {
					tagDropdown.style.width = "auto";
					tagDropdown.style.minWidth = "75px";
					tagDropdown.style.maxWidth = "300px";
					tagDropdown.style.maxHeight = "200px";
					tagDropdown.style.overflowY = "auto";

					const updateDropdown = function() {
						const value = searchInput.value;
						const cursorPosition = searchInput.selectionStart;

						// Find last "word:" pattern before cursor
						const prefixMatch = value.substring(0, cursorPosition).match(/(\w+):([^\s]*)$/);

						if (prefixMatch) {
							const category = prefixMatch[1];
							const query = prefixMatch[2];

							if (tagCategories[category]) {
								const filteredTags = Array.from(tagCategories[category])
									.filter(tag => tag.toLowerCase().startsWith(query.toLowerCase()))
									.sort();

								if (filteredTags.length > 0) {
									tagDropdown.innerHTML = "";

									filteredTags.forEach(tag => {
										const div = document.createElement("div");
										div.innerText = tag;
										div.style.padding = "5px 10px";
										div.style.cursor = "pointer";
										div.style.whiteSpace = "nowrap";

										div.onmouseover = () => div.style.background = "#f0f0f0";
										div.onmouseout = () => div.style.background = "white";

										div.onclick = () => {
											const start = prefixMatch.index + category.length + 1;
											const newValue =
												value.substring(0, start) +
												tag +
												" " +
												value.substring(cursorPosition);

											searchInput.value = newValue;
											tagDropdown.style.display = "none";
											searchInput.focus();

											const newCursorPos = start + tag.length + 1;
											searchInput.setSelectionRange(newCursorPos, newCursorPos);
										};

										tagDropdown.appendChild(div);
									});

									// Positioning
									mirror.textContent = value.substring(0, prefixMatch.index + category.length + 1);
									const textWidth = mirror.getBoundingClientRect().width;
									const inputPaddingLeft = parseFloat(getComputedStyle(searchInput).paddingLeft) || 0;

									tagDropdown.style.left = Math.min(
										textWidth + inputPaddingLeft,
										searchInput.offsetWidth - tagDropdown.offsetWidth
									) + "px";

									tagDropdown.style.display = "block";
									return;
								}
							}
						}

						tagDropdown.style.display = "none";
					};

					searchInput.addEventListener("input", updateDropdown);
					searchInput.addEventListener("click", updateDropdown);
					searchInput.addEventListener("keyup", updateDropdown);

					document.addEventListener("click", function(e) {
						if (e.target !== searchInput && e.target !== tagDropdown) {
							tagDropdown.style.display = "none";
						}
					});
				}
